## 反转链表（双指针）

```
输入: 1->2->3->4->5->NULL
输出: 5->4->3->2->1->NULL
```

```python
# -*- coding:utf-8 -*-
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None
class Solution:
    def ReverseList(self, pHead):
        pre = None
        cur = pHead
        while cur!=None:
            temp = cur.next
            cur.next = pre
            pre = cur
            cur = tmp
        return pre
```

```c++
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* pre = NULL;
        ListNode* cur = head;
        while (cur!=NULL) {
            ListNode* next = cur->next;
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        return pre;
    }
};
```
